  560.     private $ps_Check = null;

  562.     /**
  563.      * Checks whether a user has a permission or not.
  564.      *
  565.      * @param string|integer$Permission 
  566.      *             you can provide a path like /some/permission, a title, or the
  567.      *             permission ID.
  568.      *             in case of ID, don't forget to provide integer (not a string
  569.      *             containing a number)
  570.      * @param string|integer$UserID 
  571.      *             User ID of a user
  572.      *
  573.      * @throws RbacPermissionNotFoundException
  574.      * @throws RbacUserNotProvidedException
  575.      * @return boolean 
  576.      */
  577.     function check($Permission, $UserID = null)
  578.     {
  579.         if ($UserID === null)
  580.             throw new \RbacUserNotProvidedException ("\$UserID is a required argument.");
  581.  
  582.         // convert permission to ID
  583.         if (is_numeric ( $Permission ))
  584.         {
  585.             $PermissionID = $Permission;
  586.         }
  587.         else
  588.         {
  589.             if (substr ( $Permission, 0, 1 ) == "/")
  590.                 $PermissionID = $this->Permissions->pathId ( $Permission );
  591.             else
  592.                 $PermissionID = $this->Permissions->titleId ( $Permission );
  593.         }
  594.  
  595.         // if invalid, throw exception
  596.         if ($PermissionID === null)
  597.             throw new RbacPermissionNotFoundException ( "The permission '{$Permission}' not found." );
  598.  
  599.         if ($this->isSQLite())
  600.         {
  601.             $LastPart="AS Temp ON ( TR.ID = Temp.RoleID)
  602.                              WHERE
  603.                              TUrel.UserID=?
  604.                              AND
  605.                              Temp.ID=?";
  606.         }
  607.         else //mysql
  608.         {
  609.             $LastPart="ON ( TR.ID = TRel.RoleID)
  610.                              WHERE
  611.                              TUrel.UserID=?
  612.                              AND
  613.                              TPdirect.ID=?";
  614.         }
  615.         $Res=Jf::sql ( "SELECT COUNT(*) AS Result
  616.             FROM
  617.             {$this->tablePrefix()}userroles AS TUrel
  618.  
  619.             JOIN {$this->tablePrefix()}roles AS TRdirect ON (TRdirect.ID=TUrel.RoleID)
  620.             JOIN {$this->tablePrefix()}roles AS TR ON ( TR.Lft BETWEEN TRdirect.Lft AND TRdirect.Rght)
  621.             JOIN
  622.             (    {$this->tablePrefix()}permissions AS TPdirect
  623.             JOIN {$this->tablePrefix()}permissions AS TP ON ( TPdirect.Lft BETWEEN TP.Lft AND TP.Rght)
  624.             JOIN {$this->tablePrefix()}rolepermissions AS TRel ON (TP.ID=TRel.PermissionID)
  625.             ) $LastPart",
  626.             $UserID, $PermissionID );
  627.  
  628.         return $Res [0] ['Result'] >= 1;
  629.     }
  630.  
  631.     /**
  632.     * Enforce a permission on a user
  633.     *
  634.     * @param string|integer$Permission 
  635.     *             path or title or ID of permission
  636.     *
  637.     * @param integer $UserID 
  638.     *
  639.     * @throws RbacUserNotProvidedException
  640.     */
  641.     function enforce($Permission, $UserID = null)
  642.     {
  643.     if ($UserID === null)
  644.                 throw new \RbacUserNotProvidedException ("\$UserID is a required argument.");
  645.  
  646.         if (! $this->check($Permission, $UserID)) {
  647.             header('HTTP/1.1 403 Forbidden');
  648.             die("<strong>Forbidden</strong>: You do not have permission to access this resource.");
  649.         }
  650.  
  651.         return true;
  652.     }
  653.  
  654.     /**
  655.     * Remove all roles, permissions and assignments
  656.     * mostly used for testing
  657.     *
  658.     * @param boolean $Ensure 
  659.     *             must set or throws error
  660.     * @return boolean 
  661.     */
  662.     function reset($Ensure = false)
  663.     {
  664.         if ($Ensure !== true) {
  665.             throw new \Exception ("You must pass true to this function, otherwise it won't work.");
  666.             return;
  667.         }
  668.  
  669.         $res = true;
  670.         $res = $res and $this->Roles->resetAssignments ( true );
  671.         $res = $res and $this->Roles->reset ( true );
  672.         $res = $res and $this->Permissions->reset ( true );
  673.         $res = $res and $this->Users->resetAssignments ( true );
  674.  
  675.         return $res;
  676.     }
  677. }

  862.      *
  863.      * @param integer $Role 
  864.      *             ID
  865.      * @param integer $Permission 
  866.      *             ID
  867.      * @return boolean 
  868.      *
  869.      * @todo: If we pass a Role that doesn't exist the method just returns false. We may want to check for a valid Role.
  870.      */
  871.     function hasPermission($Role, $Permission)
  872.     {
  873.         $Res = Jf::sql ( "
  874.                     SELECT COUNT(*) AS Result
  875.                     FROM {$this->tablePrefix()}rolepermissions AS TRel
  876.                     JOIN {$this->tablePrefix()}permissions AS TP ON ( TP.ID= TRel.PermissionID)
  877.                     JOIN {$this->tablePrefix()}roles AS TR ON ( TR.ID = TRel.RoleID)

  884.                     SELECT parent.ID
  885.                     FROM {$this->tablePrefix()}permissions AS node,
  886.                     {$this->tablePrefix()}permissions AS parent
  887.                     WHERE node.Lft BETWEEN parent.Lft AND parent.Rght
  888.                     AND ( node.ID=? )
  889.                     ORDER BY parent.Lft
  890.                     );
  891.                     /*

  893.                     has an assignment to any of them, we're good.
  894.                     */
  895.                     ", $Role, $Role, $Permission );
  896.         return $Res [0] ['Result'] >= 1;
  897.     }

  909.     function permissions($Role, $OnlyIDs = true)

  928.                 LEFT JOIN {$this->tablePrefix()}rolepermissions AS `TR` ON (`TR`.PermissionID=`TP`.ID)
