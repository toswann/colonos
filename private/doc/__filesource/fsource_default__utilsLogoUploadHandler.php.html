  659.     protected function gd_create_scaled_image($file_name, $version, $options) {

  661.             error_log('Function not found: imagecreatetruecolor');
  662.             return false;

  672.                         $options['jpeg_quality'] : 75;

  683.                         $options['png_quality'] : 9;
  684.                 break;
  685.             default:
  686.                 return false;
  687.         }
  688.         $src_img = $this->gd_get_image_object(
  689.                 $file_path, $src_func, !empty($options['no_cache'])
  690.         );
  691.         $image_oriented = false;
  692.         if (!empty($options['auto_orient']) && $this->gd_orient_image(
  693.                         $file_path, $src_img
  694.                 )) {

  697.                     $file_path, $src_func

  701.         $max_height = $img_height = imagesy($src_img);
  702.         if (!empty($options['max_width'])) {
  703.             $max_width = $options['max_width'];
  704.         }
  705.         if (!empty($options['max_height'])) {
  706.             $max_height = $options['max_height'];
  707.         }
  708.         $scale = min(
  709.                 $max_width / $img_width, $max_height / $img_height
  710.         );
  711.         if ($scale >= 1) {
  712.             if ($image_oriented) {
  713.                 return $write_func($src_img, $new_file_path, $image_quality);
  714.             }
  715.             if ($file_path !== $new_file_path) {
  716.                 return copy($file_path, $new_file_path);
  717.             }
  718.             return true;
  719.         }
  720.         if (empty($options['crop'])) {
  721.             $new_width = $img_width * $scale;
  722.             $new_height = $img_height * $scale;
  723.             $dst_x = 0;
  724.             $dst_y = 0;
  725.             $new_img = imagecreatetruecolor($new_width, $new_height);
  726.         } else {
  727.             if (($img_width / $img_height) >= ($max_width / $max_height)) {
  728.                 $new_width = $img_width / ($img_height / $max_height);
  729.                 $new_height = $max_height;
  730.             } else {
  731.                 $new_width = $max_width;

  733.             }
  734.             $dst_x = 0 - ($new_width - $max_width) / 2;
  735.             $dst_y = 0 - ($new_height - $max_height) / 2;
  736.             $new_img = imagecreatetruecolor($max_width, $max_height);
  737.         }
  738.         // Handle transparency in GIF and PNG images:
  739.         switch ($type) {
  740.             case 'gif':
  741.             case 'png':
  742.                 imagecolortransparent($new_img, imagecolorallocate($new_img, 0, 0, 0));
  743.             case 'png':
  744.                 imagealphablending($new_img, false);
  745.                 imagesavealpha($new_img, true);
  746.                 break;
  747.         }
  748.         $success = imagecopyresampled(
  749.                         $new_img, $src_img, $dst_x, $dst_y, 0, 0, $new_width, $new_height, $img_width, $img_height
  750.                 ) && $write_func($new_img, $new_file_path, $image_quality);
  751.         $this->gd_set_image_object($file_path, $new_img);
  752.         return $success;
  753.     }
